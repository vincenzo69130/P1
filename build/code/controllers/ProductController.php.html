  460. 460     public function formatQuantityDiscounts ($specificPrices, $price, $taxRate)
  461. 461     {
  462. 462         foreach ($specificPrices as $key => &$row) {
  463. 463             $row['quantity'] = &$row['from_quantity'];
  464. 464             // The price may be directly set
  465. 465             if ($row['price'] != 0) {
  466. 466                 $cur_price = (Product::$_taxCalculationMethod == PS_TAX_EXC ? $row['price'] : $row['price'] *
  467. 467                  (1 + $taxRate / 100));
  468. 468                 if ($row['reduction_type'] == 'amount')
  469. 469                     $cur_price = Product::$_taxCalculationMethod == PS_TAX_INC ? $cur_price -
  470. 470                      $row['reduction'] : $cur_price -
  471. 471                      ($row['reduction'] / (1 + $taxRate / 100));
  472. 472                 else
  473. 473                     $cur_price = $cur_price * (1 - ($row['reduction']));
  474. 474                 $row['real_value'] = $price - $cur_price;
  475. 475             } else {
  476. 476                 global $cookie;
  477. 477                 $id_currency = (int) $cookie->id_currency;
  478. 478                 if ($row['reduction_type'] == 'amount') {
  479. 479                     $reduction_amount = $row['reduction'];
  480. 480                     if (! $row['id_currency'])
  481. 481                         $reduction_amount = Tools::convertPrice(
  482. 482                         $reduction_amount, $id_currency);
  483. 483                     $row['real_value'] = Product::$_taxCalculationMethod ==
  484. 484                      PS_TAX_INC ? $reduction_amount : $reduction_amount /
  485. 485                      (1 + $taxRate / 100);
  486. 486                 } else {
  487. 487                     $row['real_value'] = $row['reduction'] * 100;
  488. 488                 }
  489. 489             }
  490. 490             $row['nextQuantity'] = (isset($specificPrices[$key + 1]) ? (int) ($specificPrices[$key +
  491. 491              1]['from_quantity']) : - 1);
  492. 492         }
